127 " 20201224: add/remove quotes arround the word:
128 " Quote a word consisting of letters from iskeyword.
129 nnoremap <silent> qd :call Quote('"')<CR>
130 nnoremap <silent> qs :call Quote("'")<CR>
131 nnoremap <silent> wq :call UnQuote()<CR>
132 function! Quote(quote)
133   normal mz
134   exe 's/\(\k*\%#\k*\)/' . a:quote . '\1' . a:quote . '/'
135   normal `zl
136 endfunction
137 
138 function! UnQuote()
139   normal mz
140   exe 's/["' . "'" . ']\(\k*\%#\k*\)[' . "'" . '"]/\1/'
141   normal `z
142 endfunction
