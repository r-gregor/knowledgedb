Return Early Pattern

1 - Nested if statements option:

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include <stdbool.h>

boll minLetterCount(char string[], char leter, int N) {
	if (isalpha(etter)) {
		if (N >= 1 ) {
			if (strlen(string) > 0) {
				int count = 0;
				int i = 0;
				while (string[i] != '\0') {
					if (string[i] == letter) {
						count++;
					}
					i++;
				}

				if (count >= N) return true;
				else return false;

			} else {
				return false;
			}

		} else {
			return false;
		}
		
	} else {
		return false;
	}

}

int main() {

	minLetterCount("abcaad", "a", 3);

	return 0;
}


2 - Better aproach with guard clause:

#include <stdio.h>
#include <string.h>
#include <ctype.h>
#include <stdbool.h>

boll minLetterCount(char string[], char leter, int N) {

	if (!isalpha(etter))		return false;
	if (N < 1 )					return false;
	if (strlen(string) <= 0)	return false;

	int count = 0;
	int i = 0;
	while (string[i] != '\0') {
		if (string[i] == letter) {
			count++;
		}
		i++;
	}

	if (count >= N) return true;
	else return false;

}

int main() {

	minLetterCount("abcaad", "a", 3);

	return 0;
}
