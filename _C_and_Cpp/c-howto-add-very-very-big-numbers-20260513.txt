How to add very very big numbers in C
Apr 9, 2023

Generally adding two numbers is quite easy, just take two operands (number 1 and number 2) and do an
addition operation on them and desired output is there. But shown in the previous []blog (link) that there
is a limit to holding numbers.

In this case, let´s try first and check how it will go! So we will write a C code that takes two
numbers and do addition.

#include<stdio.h>

void addition(int a, int b) {
	printf("Addition of %d + %d = %d\n", a, b, a+b);
}

int main() {
	int a, b;
	a = 99999999;
	b = 1;
	addition(a, b);

	return 0;
}

And it prints as expected. Is this the real question? No! so let´s take huge numbers and try it out.

#include<stdio.h>

void addition(int a, int b) {
	printf("Addition of %d + %d = %d\n", a, b, a+b);
}

int main() {
	int a, b;
	a = 2147483647;
	b = 2147483647;

	addition(a, b);

	return 0;
}

And the output is not as expected and we need to do something so we get the expected results. In C language
integer datatype can hold max is 2147483647 and beyond this, it follows a circular path which is explained in
the previous article [**1]
[https://medium.com/@dilippuri/how-to-print-a-very-very-big-number-in-c-code-1de909573fa2](link).

Now we need to develop a different algorithm in which datatype can hold a very very large numbers and
also need to change the addition logic, we can not simply put an addition operation between operands
and get the desired result.

Before developing a new addition algorithm we need to think very fundamentally about how addition is
working. The addition is a quite
simple operation that follows below algorithm:
 * take 2 operands.
 * arrange each digit in both operands according to their place value.
 * start from the right column and add all the numbers.
 * if there is any carry, carry the number to the next place value which is next left to the current column.
 * repeat the same pattern for each remaining column.

Check the complete addition tutorial here
[https://content.byui.edu/file/b8b83119-9acc-4a7b-bc84-efacf9043998/1/Math-1-3-2.html](link). Now we
understood the addition process, we can follow a similar approach for very big numbers.

Let´s take three stacks of integers which will hold two operands and a result stack. We will pop a
digit from each stack and add them and store the digit in the result stack, in case of any carry we will
forward it to the next column operation. We will same process until each stack become empty.

#include<stdio.h>
#include<limits.h>
#include<stdlib.h>

#define MAX(a, b) (a > b ? a : b)
#define MIN(a, b) (a < b ? a : b)

struct number_stack {
	int top;
	unsigned stack_height;
	int* array;
};

/* Stack creation function */
struct number_stack* createNumberStack(unsigned stack_height) {
	struct number_stack* numberStack =
		(struct number_stack*) malloc(sizeof(struct number_stack));
	numberStack->top = -1;
	numberStack->stack_height = stack_height;
	numberStack->array =
		(int*) malloc(numberStack->stack_height * sizeof(int));

	return numberStack;
}

/* Stack is full when the top is equal to the last index */
int isFull(struct number_stack* numberStack) {
	return numberStack->top == (numberStack->stack_height - 1);
}

/* Stack is empty when the top is equal to -1 */
int isEmpty(struct number_stack* numberStack) {
	return numberStack->top == -1;
}

/* Push function to add an element on top of the stack and
 * top increases be 1
 */
void push(struct number_stack* numberStack, int item) {
	if (isFull(numberStack))
		return;
	numberStack->array[++numberStack->top] = item;
}

/* Pop function to remove an element from the top of stack
 * and top decreases by 1
 */
int pop(struct number_stack* numberStack) {
	if (isEmpty(numberStack))
		return INT_MIN;
	return numberStack->array[numberStack->top - ];
}

int main() {
	// two very very big numbers
	char number_1_str[] =
		"01234567890123456789012345678901234567890123456789012345678901234567890123456789";
	char number_2_str[] =
		"98765432109876543210987654321098765432109876543210987654321098765432109876543210";
	int number_1_str_len, number_2_str_len, result_stack_len;

	number_1_str_len = sizeof(number_1_str)-1;
	number_2_str_len = sizeof(number_2_str)-1;
	result_stack_len = MAX(number_1_str_len, number_2_str_len);

	// creating stacks for each operand and result
	struct number_stack* number_stack1 = createNumberStack(number_1_str_len);
	struct number_stack* number_stack2 = createNumberStack(number_2_str_len);
	struct number_stack* result_stack = createNumberStack(result_stack_len+1);

	// push number 1 to stack 1
	for(int i = 0; i < number_1_str_len; i++) {
		push(number_stack1, number_1_str[i] - '0');
	}

	// push number 2 to stack 2
	for(int i = 0; i < number_2_str_len; i++) {
		push(number_stack2, number_2_str[i] - '0');
	}

	int temp_result_stack_len = MIN(number_1_str_len, number_2_str_len);
	int temp, temp1, temp2;
	int carry = 0;

	// addition of number 1 and number 2
	for(int i = 0; i < temp_result_stack_len; i++) {
		temp1 = pop(number_stack1);
		temp2 = pop(number_stack2);
		temp = temp1 + temp2 + carry;
		carry = temp/10;
		temp = temp%10;
		push(result_stack, temp);
	}

	// carry forwarding if number 1 is greater
	if (number_1_str_len > number_2_str_len) {
		for(int i = 0; i < result_stack_len - temp_result_stack_len; i++) {
			temp1 = pop(number_stack1);
			temp = temp1 + carry;
			carry = temp/10;
			temp = temp%10;
			push(result_stack, temp);
		}
	}

	// carry forwarding if number 2 is greater
	if (number_1_str_len < number_2_str_len) {
		for(int i = 0; i < result_stack_len - temp_result_stack_len; i++) {
			temp2 = pop(number_stack2);
			temp = temp2 + carry;
			carry = temp/10;
			temp = temp%10;
			push(result_stack, temp);
		}
	}

	// last digit carry
	push(result_stack, carry);

	printf("Number 1: %s\n", number_1_str);
	printf("Number 2: %s\n", number_2_str);
	printf("Addition of Number1 and Number2 is: ");

	// printing results
	for(int i = 0; i < result_stack_len; i++) {
		printf("%d", pop(result_stack));
	}

	printf("\n");

	return 0;
}

Now lets compile and run the code.

cc big_number_addition.c
./a.out

And here is the output as expected.

Number 2: 98765432109876543210987654321098765432109876543210987654321098765432109876543210
Addition of Number1 and Number2 is:
099999999999999999999999999999999999999999999999999999999999999999999999999999999

How to print a very very big number in C
Apr 8, 2023

Printing number in C is a very easy task, indeed it´s easy, it is kind of hello world code in C. So
let´s write a code which takes a number and print it.

/******************************************************************/
#include<stdio.h>
void print_big_number(int number) {
	printf("Number is: %d\n", number);
}

int main() {
	int number;
	printf(" - - - - - - - - - - - - - - - - - - - \n");
	number = 47;
	print_big_number(number);
	printf(" - - - - - - - - - - - - - - - - - - - \n");

	return 0;
}
/******************************************************************/

Above code taking an integer(of course easy one 😉) and just printing it. Let´s compile and run it.

$> gcc print_big_number.c
#to run
$> ./a.out

Output:

Yaay! number got printed, very happy. Let´s now try with little bigger number.

So what if we wants to print greater than 2147483647. Till now you might have noticed that we are using
datatype as "int" means signed integer which has the capacity to hold the numbers which shown in
above diagram. For greater numbers we can use other datatypes which can hold very big numbers. So here
is the list of ranges for each datatype.
