Reading a CSV in C and storing it in a multidimensional array requires deciding:
  (1) the data type (strings vs numbers),
  (2) whether the grid is fixed-size or dynamic, and
  (3) how to handle quoted fields and separators.

Below are two practical approaches with example code: a fixed-size numeric grid (simpler, fast) and a dynamic
string-grid that handles quoted fields and arbitrary sizes.

1) Fixed-size numeric CSV (rows and cols known or bounded)
    Simple, fast, minimal dependencies.
    Assumes no quoted commas and every cell is convertible to double/int.

Example: read up to MAX_ROWS × MAX_COLS doubles.

#include <stdio.h>
#include <stdlib.h>

#define MAX_ROWS 1000
#define MAX_COLS 1000

int main(void) {
	const char *filename = "data.csv";
	FILE *f = fopen(filename, "r");
	if (!f) { perror("fopen"); return 1; }

	double grid[MAX_ROWS][MAX_COLS];
	size_t rows = 0, cols = 0;

	char line[8192];
	while (fgets(line, sizeof line, f) && rows < MAX_ROWS) {
		size_t cur_col = 0;
		char *p = line;
		char *token;
		while ((token = strsep(&p, ",\n")) != NULL && cur_col < MAX_COLS) {
			if (*token == '\0') {
				grid[rows][cur_col++] = 0.0; // treat empty as zero (customize as needed)
			} else {
				grid[rows][cur_col++] = strtod(token, NULL);
			}
		}
		if (cur_col > cols) cols = cur_col;
		rows++;
	}

	fclose(f);

	// Example: print grid
	for (size_t r = 0; r < rows; ++r) {
		for (size_t c = 0; c < cols; ++c) {
			printf("%.6g%s", grid[r][c], (c + 1 == cols) ? "\n" : ", ");
		}
	}

	return 0;
}

Notes:
    strsep is GNU; on non-GNU systems you can implement a simple tokenizer using strchr and manual slicing.
    Error handling, trimming whitespace, or locale-aware parsing may be added.

2) Dynamic string grid with proper CSV parsing (variable rows/cols, quoted fields)
    Handles quoted fields (commas inside quotes), doubled quotes, varying row lengths.
    Stores pointers to heap-allocated C strings in a dynamically resized 2D array.

Example: a compact CSV parser that returns a vector of rows, each row is a vector of char *.

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

typedef struct {
	char **cells;
	size_t n;
} Row;

typedef struct {
	Row *rows;
	size_t n;
} CSV;

static void row_init(Row *r) { r->cells = NULL; r->n = 0; }

static void row_push(Row *r, char *cell) {
	r->cells = realloc(r->cells, (r->n + 1) * sizeof(*r->cells));
	r->cells[r->n++] = cell;
}

static void row_free(Row *r) {
	for (size_t i = 0; i < r->n; ++i) free(r->cells[i]);
	free(r->cells);
	r->cells = NULL; r->n = 0;
}
static void csv_init(CSV *c) { c->rows = NULL; c->n = 0; }

static void csv_push(CSV *c, Row *r) {
	c->rows = realloc(c->rows, (c->n + 1) * sizeof *c->rows);
	c->rows[c->n++] = *r;
}

static void csv_free(CSV *c) {
	for (size_t i = 0; i < c->n; ++i) row_free(&c->rows[i]);
	free(c->rows);
	c->rows = NULL; c->n = 0;
}

/* Parse input stream as CSV using RFC-4180-like rules:
- Fields separated by commas
- Fields may be quoted with " and can contain commas, newlines
- Doubled quotes "" inside quoted fields => one " */
CSV parse_csv(FILE *f) {
	CSV csv; csv_init(&csv);
	Row row; row_init(&row);

	int ch;
	char *buf = NULL;
	size_t cap = 0, len = 0;
	int in_quotes = 0;
	int started = 0;

	while ((ch = fgetc(f)) != EOF) {
		if (!started) { started = 1; in_quotes = 0; len = 0; }

		if (in_quotes) {
			if (ch == '"') {
				int nxt = fpeek(f); // helper below
				if (nxt == '"') {
					fgetc(f); // consume second quote
					if (len + 1 >= cap) { cap = cap ? cap * 2 : 64; buf = realloc(buf, cap); }
					buf[len++] = '"';
				} else {
					in_quotes = 0; // closing quote
				}
			} else {
				if (len + 1 >= cap) { cap = cap ? cap * 2 : 64; buf = realloc(buf, cap); }
				buf[len++] = ch;
			}
		} else {
			if (ch == '"') {
				in_quotes = 1;
			} else if (ch == ',') {
				// end field
				if (len + 1 >= cap) { cap = cap ? cap * 2 : 64; buf = realloc(buf, cap); }
				buf[len] = '\0';
				row_push(&row, strdup(buf));
				len = 0;
			} else if (ch == '\r') {
				// ignore CR, handle LF in next iteration
			} else if (ch == '\n') {
				if (len + 1 >= cap) { cap = cap ? cap * 2 : 64; buf = realloc(buf, cap); }
				buf[len] = '\0';
				row_push(&row, strdup(buf));
				csv_push(&csv, &row);
				row_init(&row);
				len = 0;
			} else {
				if (len + 1 >= cap) { cap = cap ? cap * 2 : 64; buf = realloc(buf, cap); }
				buf[len++] = ch;
			}
		}
	}

	// last field/row if file doesn´t end with newline
	if (started) {
		if (in_quotes) {
			// unterminated quotes: accept accumulated; could add error handling
		}
		if (len + 1 >= cap) { cap = cap ? cap * 2 : 64; buf = realloc(buf, cap); }
		buf[len] = '\0';
		row_push(&row, strdup(buf));
		csv_push(&csv, &row);
	}
	free(buf);
	return csv;
}

/* Small helper: peek next char in FILE* without consuming (uses fpos)
If fseek/fgetpos not available on some streams, adjust accordingly. */
static int fpeek(FILE *f) {
	int c = fgetc(f);
	if (c == EOF) return EOF;
	ungetc(c, f);
	return c;
}

/* Usage example */
int main(void) {
	FILE *f = fopen("data.csv", "r");
	if (!f) { perror("fopen"); return 1; }

	CSV csv = parse_csv(f);
	fclose(f);

	// Print CSV and free memory
	for (size_t r = 0; r < csv.n; ++r) {
		for (size_t c = 0; c < csv.rows[r].n; ++c) {
			printf("[%zu,%zu]=\"%s\"%s", r, c, csv.rows[r].cells[c],
				c + 1 == csv.rows[r].n ? "\n" : ", ");
		}
	}

	csv_free(&csv);
	return 0;
}

Notes and recommendations
For production parsing prefer a tested library (libcsv, csvparser, or use higher-level languages) if
performance and full RFC compliance are required.  Memory: dynamic approach stores each cell separately; for
large files consider storing contiguous buffers or parsing on-the-fly to avoid high memory usage.  Numeric
conversion: after reading strings, convert fields with strtod/strtol and check errors (errno, end pointer).
Edge cases: different separators (semicolon), escaped newlines, different line endings, and encodings (UTF-8)
require small adjustments.

Summary checklist
If structure is fixed and values are numeric: use fixed-size arrays and simple tokenization for speed.  If
CSVs are arbitrary and may contain quotes/newlines: use a proper parser (example above) or an available CSV
library.  Always handle memory allocation failures and validate numeric conversions before using values.

How can we store values into an array of characters in the C programming language?

There are many ways to do so.

Of course, you need space to store it. At least enough space for the larger possible string + 0 (end of
string). Remember strings in C ends where the 0 is found, but not strings variables....

The first thing you need is to understand the difference between an array and a pointer

char *p; // is a pointer
char s[512]; //is an array of char

Watch that s can hold up to 511 chars + 0, no matter if it´s holding only "Hello". That is "the reserved space

Whenever you recall s you get a pointer to the first element of s (it is the same than &s[0]), but the
difference with p that is also a pointer to char, is that while s has a reserved space that is pointing, p
doesn´t so. So you can´t store chars in p unless it points to a reserved space, like, i.e, s or a point in s,
for example

p=&s[20];

On the other hand, you can change the address that p is pointing, but you must never do that with s, otherwise
you will loose the entry point of the reserved array.

The last thing you know to understand before learning how to store values in an char array, is constants
strings.

In c, you can define a constant string enclosing it in " "

const char *msg="Hello World";

msg will also return a pointer like s and p but it would be a const char* , and will have a reserved space to
store the message like s, but it content will and must remain unchanged. C will never change it, and
programmer must never change it (yes, you can change it, but don´t do it). That is what word "const" means as
a different kind of pointer.

You can point p to message, even at p declaration, but it is unnecessary and, might be dangerous if you don´t
exactly know what you are doing.

About differences between a const char pointer and a char pointer: a function that accepts a constant char
pointer, like int puts(const char*s) can admit constant strings, pointer to constants strings, or pointers to
strings:

//all valid options:
puts("hello world ");
puts(msg);
puts(p);
puts(s);

//even
puts(gets(s));

but if a function parameter accepts only char * (without const keyword, that means that the contents of the
pointed string can be changed, and only arguments like p and s or &s[20] are admitted.

The other difference between the reserved space of s and msg is that the one of msg has the exactly size to
keep the message + an extra space to the string terminator (end of string), the zero char. While the reserved
space of s might be equal or greater than the stored string (length +1). In the above example it has 512 chars
space length, that means that can store strings up to 511 + 1 length. Always remember the space for 0, the
string terminator.

So, we learned about:
  * string: sequence of chars + a 0 at the end
  * constant strings: unchangeable strings
  * char array: reserved space to hold strings
  * char pointers and const char pointers

Now, that you understand chars strings, lets see the many ways you have to store values to a string:

User/file entry:
In C, there is (almost) not difference between a file or console. They are both streams. What a user types in
a keyboard is sent to C using a stream called stdin (standard in) already opened in stdio.h

Some examples here, using only stdin

scanf("%s", s);
fscanf(stdin,"%s", s);

notice that scanf is only a macro of fscanf

lets see another function:
fgets(s, 511, stdin);

Notice:
Here you can enter the space reserved, to avoid reading more chars than the available space (security) I
didn´t gave an example using gets function. It exists but is considered insecure because can not control the
amount of entered chars.

Unintentionally or intentionally a user could break yor program entering more characters than the available.
That is why fgets should always be used instead of gets, and that´s why I got to explain about streams

Now that I already mentioned streams and files , lets show you how to fill your string using a text file.

The text file are not automatically open. You need your own file descriptor (like stdin but for your file),
and need to opem it as "rt" (reading text) or simply "r".

#include<stdio.h>

FILE *file // file descriptor
char line[512]; //holding space

int main() {
	//open file
	file = fopen("myFile.txt","rt");

	//if file couldn´t be opened, ...
	if( file == 0)
		puts("Couldn´t open file");
	else {
		fgets(line,511,file);
		puts("First line of file:");
		puts(line);
		fclose(file);
	}

	return 0;
}

Of course, it reads only one line. To read all use a while (!feof(file)) loop

Copy from other string (variable or constant) to a reserved space:
string.h gives you tools to copy a string to a char array (reserved space). The original string could be a
constant or a string. Lets say you want to copy msg to s

the function prototype is

#include <string.h>

char *strncpy(char *dest, const char *src, size_t n);

Notice how destination is char* because it gonna be modified, while source is const char*, that means that you
can use a constant string or a pointer or anything, but they also say "it won´t be changed ".

there is also a strcp (without size restriction) but for security reasons, it shouldn´t be used.

So, assuming you already included string.h, any of these lines could be performed:

strncpy(s,msg,511);
strncpy(s,"Hello World",511);

or, to concatenate use strncat but be careful, here the size n is the space left by the source string, and
must be calculated using strlen(s);

strlen will read the length of the string stored in s, not the 512 reserved space.

strncpy(s, "Hello", 511);
strncat(s, " World", 511-strlen(s));
strncat(s, " !!!!", 511-strlen(s));

Of course, if you don´t want to use string.h functions, you can always use the old school method

int i;
for(i=0; msg[i]!=0 && i<512; i++) {
	s[i]=msg[i];
}
s[i]=0; //add string terminator

and it will run faster.

Creating a Duplicate of an existing string:
string.h also gives you a function to create a duplicate of an existing string with the advantages that, the
reserved space is created on the fly (so you don´t need an array declared to store it), and that its size is
exactly the size of string + 1.

strdup(p,msg);

that makes a copy of msg that is not a constant. But, be careful, size of the copy is size of msg, that means
you can´t concatenate and you should assign it others values, specially larger than the original. Yes you can
modify its char like changing "-" by ".", or uppercasing it, or reversing...

Converting numbers or other printings to a string
stdio.h give us "string" version of io functios. They are useful to convert a number to / from string.

sprintf(s,"%g",3.1415926);

//you can set format
//or add text to that
double height = 12.342;
sprintf(s, " h = %4.1g m.", height );

Or get numbers from strings

int k;
const char *value="5";
sscanf(value,"%d", &k);

// other example
const char *time = "12:45:01";
int hh,mm,ss;
sscanf(time,"%d:%d:%d", &hh,&mm,&ss);

Misbehaving:
Because of strings are hold in arrays, they are very transparent to C programmers and you can find a lot of
misbehaving ways to change its contents. Be free and assume the consequences. It is C.
